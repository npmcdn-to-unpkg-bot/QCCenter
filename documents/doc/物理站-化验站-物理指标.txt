CREATE VIEW [dbo].[View_Print_HeCha]
AS
SELECT     a.ProduceDate AS 生产日期, a.CartNumber AS 车号, b.ProductName AS 品种, a.goodrate AS 好品率, a.ErrPicCount AS 缺陷条数
FROM         (SELECT     ProduceDate, CartNumber, ProductType, GoodRate * 100 AS goodrate, ErrPicCount
                       FROM          dbo.CartData
                       UNION
                       SELECT     ProduceDate, CartNumber, ProductTypeID AS producttype, GoodRate, ErrPicCount
                       FROM         dbo.MaHouData) AS a INNER JOIN
                      dbo.ProductData AS b ON a.ProductType = b.ProductID
